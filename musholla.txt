    <script>
        // --- DATA & STATE MANAGEMENT ---
        // Data awal (Default) jika belum ada di LocalStorage
        const defaultData = {
            bkm: {
                name: "BKM Al Jihad",
                desc: "Badan Kemakmuran Musholla Al Jihad berdedikasi untuk memakmurkan musholla dan membina generasi muda melalui kegiatan keagamaan dan sosial.",

            },
            mentor: {

                photo: "https://picsum.photos/seed/khairi/300/300"
            },
            rajMembers: [
                { id: 1, name: "Siti Aminah", role: "Sekretaris", photo: "https://picsum.photos/seed/siti/200/200" },
                { id: 2, name: "Rizky Pratama", role: "Bendahara", photo: "https://picsum.photos/seed/rizky/200/200" },
                { id: 3, name: "Ahmad Dani", role: "Ketua Remaja", photo: "https://picsum.photos/seed/dani/200/200" },
                { id: 4, name: "Dewi Sartika", role: "Koordinator Acara", photo: "https://picsum.photos/seed/dewi/200/200" }
            ],
            guests: [] // Menyimpan data tamu yang check-in
        };

        // Load data dari LocalStorage atau gunakan default
        let appData = JSON.parse(localStorage.getItem('muharramApp_V3')) || defaultData;

        // Fungsi Simpan Data
        function saveAppData() {
            localStorage.setItem('muharramApp_V3', JSON.stringify(appData));
            renderAll();
        }

        // --- RENDER FUNCTIONS ---
        function renderAll() {
            renderBKM();
            renderMentor();
            renderRAJ();
            renderGuests();
        }

        function renderBKM() {
            document.getElementById('bkm-name').innerText = appData.bkm.name;
            document.getElementById('bkm-desc').innerText = appData.bkm.desc;
            document.getElementById('bkm-logo').src = appData.bkm.logo;
        }

        function renderMentor() {
            document.getElementById('mentor-name').innerText = appData.mentor.name;
            document.getElementById('mentor-photo').src = appData.mentor.photo;
        }

        function renderRAJ() {
            const grid = document.getElementById('raj-grid');
            grid.innerHTML = '';
            appData.rajMembers.forEach(m => {
                const div = document.createElement('div');
                div.className = 'member-card';
                div.innerHTML = `
                    <div class="card-actions">
                        <button class="icon-btn" onclick="editMember(${m.id})"><i class="ph ph-pencil-simple"></i></button>
                        <button class="icon-btn del" onclick="deleteMember(${m.id})"><i class="ph ph-trash"></i></button>
                    </div>
                    <img src="${m.photo}" class="member-avatar">
                    <span class="member-badge">${m.role}</span>
                    <h4 style="margin-top:10px;">${m.name}</h4>
                `;
                grid.appendChild(div);
            });
        }

        function renderGuests() {
            const list = document.getElementById('guest-list');
            list.innerHTML = '';
            
            if (appData.guests.length === 0) {
                list.innerHTML = '<p class="text-center" style="padding: 20px; color: #999;">Belum ada data tamu.</p>';
                return;
            }

            appData.guests.forEach(g => {
                let badgeClass = 'bg-tamu';
                if(g.role === 'Remaja Masjid') badgeClass = 'bg-remaja';
                if(g.role === 'Jamaah') badgeClass = 'bg-jamaah';

                const div = document.createElement('div');
                div.className = 'guest-item';
                div.innerHTML = `
                    <div class="guest-info">
                        <strong>${g.name}</strong>
                        <div class="guest-meta">${g.origin}</div>
                    </div>
                    <div class="flex items-center gap-4">
                        <span class="guest-badge ${badgeClass}">${g.role}</span>
                        <span style="font-size:0.8rem; color:#999;">${g.time}</span>
                        <button onclick="deleteGuest(${g.id})" style="color:red; background:none; border:none; cursor:pointer;"><i class="ph ph-trash"></i></button>
                    </div>
                `;
                list.appendChild(div);
            });
        }

        // --- LOGIC: BKM & MENTOR ---
        function updateLogo(input) {
            if (input.files && input.files[0]) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    appData.bkm.logo = e.target.result;
                    saveAppData();
                }
                reader.readAsDataURL(input.files[0]);
            }
        }

        function editBKMInfo() {
            document.getElementById('info-name').value = appData.bkm.name;
            document.getElementById('info-desc').value = appData.bkm.desc;
            document.getElementById('info-modal').style.display = 'flex';
        }

        function updateMentorPhoto(input) {
            if (input.files && input.files[0]) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    appData.mentor.photo = e.target.result;
                    saveAppData();
                }
                reader.readAsDataURL(input.files[0]);
            }
        }

        function editMentorName() {
            const newName = prompt("Masukkan Nama Pembina Baru:", appData.mentor.name);
            if(newName && newName.trim() !== "") {
                appData.mentor.name = newName;
                saveAppData();
            }
        }

        function saveInfo() {
            appData.bkm.name = document.getElementById('info-name').value;
            appData.bkm.desc = document.getElementById('info-desc').value;
            saveAppData();
            closeModal('info-modal');
        }

        // --- LOGIC: RAJ MEMBERS ---
        function openMemberModal(id = null) {
            const modal = document.getElementById('member-modal');
            document.getElementById('edit-id').value = id || '';
            const nameIn = document.getElementById('member-name');
            const roleIn = document.getElementById('member-role');
            const photoIn = document.getElementById('member-photo');

            if (id) {
                const m = appData.rajMembers.find(x => x.id === id);
                nameIn.value = m.name;
                roleIn.value = m.role;
                photoIn.value = ""; 
            } else {
                nameIn.value = "";
                roleIn.value = "";
                photoIn.value = "";
            }
            modal.style.display = 'flex';
        }

        function saveMember() {
            const id = document.getElementById('edit-id').value;
            const name = document.getElementById('member-name').value;
            const role = document.getElementById('member-role').value;
            const file = document.getElementById('member-photo').files[0];

            if(!name) { alert("Nama harus diisi!"); return; }

            if (id) {
                // Edit
                const m = appData.rajMembers.find(x => x.id == id);
                m.name = name;
                m.role = role;
                if(file) {
                    const reader = new FileReader();
                    reader.onload = (e) => { m.photo = e.target.result; saveAppData(); closeModal('member-modal'); };
                    reader.readAsDataURL(file);
                    return;
                }
            } else {
                // Add New
                const newId = Date.now();
                let photoUrl = `https://picsum.photos/seed/${newId}/200/200`;
                const newMember = { id: newId, name, role, photo: photoUrl };
                
                if(file) {
                    const reader = new FileReader();
                    reader.onload = (e) => { 
                        newMember.photo = e.target.result; 
                        appData.rajMembers.push(newMember); 
                        saveAppData(); 
                        closeModal('member-modal');
                    };
                    reader.readAsDataURL(file);
                    return;
                }
                appData.rajMembers.push(newMember);
            }
            saveAppData();
            closeModal('member-modal');
        }

        function editMember(id) { openMemberModal(id); }
        function deleteMember(id) {
            if(confirm("Hapus anggota ini?")) {
                appData.rajMembers = appData.rajMembers.filter(m => m.id !== id);
                saveAppData();
            }
        }

        // --- LOGIC: CHECK-IN GUESTS ---
        function handleCheckIn(e) {
            e.preventDefault();
            const name = document.getElementById('guest-name').value;
            const origin = document.getElementById('guest-origin').value;
            const role = document.getElementById('guest-role').value;
            
            const now = new Date();
            const timeString = now.getHours().toString().padStart(2, '0') + ':' + now.getMinutes().toString().padStart(2, '0');

            appData.guests.unshift({
                id: Date.now(),
                name,
                origin,
                role,
                time: timeString
            });

            document.getElementById('checkin-form').reset();
            saveAppData();
            
            // Efek visual (scroll ke daftar)
            document.getElementById('guest-list').scrollTop = 0;
        }

        function deleteGuest(id) {
            if(confirm("Hapus data tamu ini?")) {
                appData.guests = appData.guests.filter(g => g.id !== id);
                saveAppData();
            }
        }

        function triggerUpload(id) { document.getElementById(id).click(); }
        function closeModal(id) { document.getElementById(id).style.display = 'none'; }
        
        window.onclick = function(e) {
            if(e.target.classList.contains('modal')) e.target.style.display = 'none';
        }

        // --- COUNTDOWN ---
        const targetDate = new Date("June 28, 2026 00:00:00").getTime();
        setInterval(() => {
            const now = new Date().getTime();
            const dist = targetDate - now;
            if(dist > 0) {
                document.getElementById('d').innerText = Math.floor(dist / (1000*60*60*24));
                document.getElementById('h').innerText = Math.floor((dist % (1000*60*60*24)) / (1000*60*60));
                document.getElementById('m').innerText = Math.floor((dist % (1000*60*60)) / (1000*60));
                document.getElementById('s').innerText = Math.floor((dist % (1000*60)) / 1000);
            }
        }, 1000);

        // Init
        renderAll();

    </script>
